    # main.js 中常用的一些方法
    // // 刷新保存状态管理
if (sessionStorage.getItem('store')) {
  store.replaceState(
    Object.assign(
      {},
      store.state,
      JSON.parse(sessionStorage.getItem('store'))
    )
  )
  sessionStorage.removeItem('store')
}

// 监听，在页面刷新时将vuex里的信息保存到sessionStorage里
window.addEventListener('beforeunload', () => {
  sessionStorage.setItem('store', JSON.stringify(store.state))
})


//引入全局过滤器

// 导出的是对象，可以直接通过 key 和 value 来获得过滤器的名和过滤器的方法
Object.keys(custom).forEach(key => {
  Vue.filter(key, custom[key])
})

    # vue中优化资源加载

    1、 js使用docomuent.querySelector('xxx')获取某一个dom节点， 2、 但在vue中可以使用this.$refs.xxx获取某一个节点。$refs是vue虚拟出的dom，可以有效较少获取/操作DOM节点的性能消耗。 3、 打包时使用gzip压缩可以减少资源加载 4、 取消预加载资源的方式,在vue.config.js文件添加一下代码

    chainWebpack: config => {
        config.plugins.delete('prefetch')
    },
    # vue的引入方式

    1、如果是一个方法，要写成

      import {xxx} from '@/utils/xxx'

    2、如果是项目中指定的某个组件，应当写成

      import xxx from '@/components/xxx'

    两者的区别是一个有花括号，一个没有花括号

    3、当组件是从第三方多个引入的时候，写法和第一种类似，是要添加花括号的。

    # watch、methods、computed三者的关系
    // computed
<!-- {{}}中数据先从data中找，找不到再到computed中找 -->
<div id="#app">{{fullName}} {{age}}</div>
<script>
var vm = new Vue({
	el:'#app',
	data:{
		firstName:'1',
		lastName:'2',
		age:30
	},
	// 内置缓存,只有firstName和lastName数据变了，才执行一次
	//age变了不会执行
	computed:{
		fullName(){
			return this.firstName+this.lastName
		}
	}
})
</script>

// methods,要用()方式调用才可
<div id="#app">{{fullName()}} {{age}}</div>
<script>
var vm = new Vue({
	el:'#app',
	data:{
		firstName:'1',
		lastName:'2',
		age:30
	},
	//  // 页面只要重新渲染,methods里的方法就会重新执行
	//age变仍会执行
	methods:{
		fullName(){
			return this.firstName+this.lastName
		}
	}
})
</script>


// watch
<div id="#app">{{fullName}} {{age}}</div>
<script>
var vm = new Vue({
	el:'#app',
	data:{
		firstName:'1',
		lastName:'2',
		fullName:'', //要有初始值
		age:30
	},
	//首次渲染页面不会执行
	// 内置缓存,只有firstName和lastName数据变了，才执行一次
	//age变了不会执行
	//写起来比较麻烦
	watch:{
		firstName(){
			this.fullName =this.firstName+this.lastName
		},
		lastName(){
			this.fullName = this.firstName+this.lastName
		}
	}
})
</script>
