let name1 = nameArr[0];

# 判断是否以某一字符串开头或者以某一字符串结尾、包含

//判断结尾

# 深拷贝与浅拷贝
  `Object.asign()拷贝(如果是对象中含有对象，那么拷贝的就是浅拷贝，如果只是一个单层的对象，那么就可以理解为深拷贝。)`
    let obj1 = { dog1: { name: '小煤球' } };
    let obj2 = { name: '小Python' };
    let obj = {};

    Object.assign(obj, obj1, obj2);
    console.log(obj);
  `延展拷贝(浅拷贝)`
    let obj1 = { dog1: { name: '小煤球' } };
    let obj2 = { name: '小Python' };
    console.log({ ...obj1, ...obj2 });

  `JSON.parse和JSON.stringify深拷贝`
    let obj1 = { dog1: { name: '小煤球' } };
    let obj2 = { name: '小Python' };
    console.log(JSON.parse(JSON.stringify({ ...obj1, ...obj2 })));
